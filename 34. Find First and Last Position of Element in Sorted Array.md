# Find First and Last Position of Element in Sorted Array
## 方法 二分法
分别做两次二分查找找到左边界和右边界
## 复杂度
time: O(logn)
## 代码
```java
class Solution {
    public int[] searchRange(int[] nums, int target) {
        int left = 0, right = nums.length -1;
        int [] res = {-1, -1};
        if (nums== null || nums.length == 0) {
            return res;
        }
        //find left
        while (left + 1< right) {
            int mid = left +(right -left)/2;
            if (nums[mid] <target) {
                left = mid;
            } else if (nums[mid] >= target) {
                right = mid;
            }
        }
        if (nums[left]== target) {
             res[0] = left;
        } else if (nums[right]== target) {
             res[0] = right;
        } else {
            return res;
        }

        //find right
        left = 0;
        right = nums.length -1;
        while (left + 1< right) {
            int mid = left +(right -left)/2;
            if (nums[mid] <=target) {
                left = mid;
            } else if (nums[mid] > target) {
                right = mid;
            }
        }
        if (nums[right]== target) {
             res[1] = right;
        } else {
             res[1] = left;
        }

        return res;
    }
}
```
